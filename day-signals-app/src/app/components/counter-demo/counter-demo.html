Implementation

count = signal(0);
step = signal(1);

// Computed signals automatically update
doubled = computed(() => count() * 2);
isEven = computed(() => count() % 2 === 0);
isPositive = computed(() => count() > 0);

summary = computed(() => {
  const type = isEven() ? 'even' : 'odd';
  const sign = isPositive() ? 'positive' : 'negative';
  return `${count()} is ${type} and ${sign}`;
});

increment() {
  count.update(c => c + step());
}
